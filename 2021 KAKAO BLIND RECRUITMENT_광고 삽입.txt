// 2021 KAKAO BLIND RECRUITMENT : 광고 삽입 (구간합 + 투포인터)
#include <string>
#include <vector>
#include <queue>

using namespace std;

// 해당 영상의 초당 시청자의 수
int viewerPerSec[360000];

// 시간 문자열을 초 단위 시간으로 변환하는 함수
int convertStringToSecond(string str) {
    int h = stoi(str.substr(0, 2)) * 3600;
    int m = stoi(str.substr(3, 2)) * 60;
    int s = stoi(str.substr(6, 2));

    return h + m + s;
}

// 초 단우 시간을 시간 문자열로 변환하는 함수
string convertSecondToString(int sec) {
    string result = "";

    int h = sec / 3600;
    sec -= (h * 3600);
    int m = sec / 60;
    sec -= (m * 60);
    int s = sec;

    if (h < 10) result += "0";
    result += (to_string(h) + ":");

    if (m < 10) result += "0";
    result += (to_string(m) + ":");

    if (s < 10) result += "0";
    result += to_string(s);

    return result;
}

string solution(string play_time, string adv_time, vector<string> logs) {
    string answer = "";

    // 전체 영상의 초당 시청자의 수를 구한다.
    for (string log : logs) {
        int startTime = convertStringToSecond(log.substr(0, 8));
        int endTime = convertStringToSecond(log.substr(9));

        for (int i = startTime; i < endTime; i++) {
            viewerPerSec[i]++;
        }
    }

    int playTime = convertStringToSecond(play_time);
    int adTime = convertStringToSecond(adv_time);

    // 초당 시청자의 수가 가장 많은 광고 재생시간 길이 만큼의 구간(구간합)을 구한다.
    queue<int> q;
    int startTime = 0; // 구간합의 시작 위치
    long long maxSum = 0; // 구간합의 최대
    long long sum = 0; // 구간합
    for (int i = 0; i < adTime; i++) {
        sum += viewerPerSec[i];
        q.push(viewerPerSec[i]);
    }
    maxSum = sum; // 구갑합의 최대 갱신

    // 1초씩 앞으로 구간 이동
    for (int i = adTime; i < playTime; i++) {
        sum -= q.front();
        q.pop();

        sum += viewerPerSec[i];
        q.push(viewerPerSec[i]);

        if (sum > maxSum) {
            maxSum = sum;
            startTime = i - adTime + 1;
        }
    }

    answer = convertSecondToString(startTime);

    return answer;
}
