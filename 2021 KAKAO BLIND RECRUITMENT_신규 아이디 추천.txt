#include <string>
#include <vector>

using namespace std;

string solution(string new_id) {
    string ans;

    // 1~3 단계 처리
    for (int i = 0; i < new_id.size(); i++) {
        // 소문자 또는 숫자인 경우
        if (('a' <= new_id[i] && new_id[i] <= 'z') || 
            ('0' <= new_id[i] && new_id[i] <= '9')) continue;

        if ('A' <= new_id[i] && new_id[i] <= 'Z') { // 대문자인 경우
            new_id[i] += 32;
        }
        else if (new_id[i] != '-' && new_id[i] != '_' && new_id[i] != '.') { // 사용 불가능한 특수문자인 경우
            new_id.erase(new_id.begin() + i);
            i--;
        }
        else if (new_id[i] == '.' && (i-1 >= 0 && new_id[i-1] == '.')) {
            new_id.erase(new_id.begin() + i);
            i--;
        }
    }
    
    // 4단계 처리
    if (new_id[0] == '.') new_id.erase(new_id.begin());
    if (!new_id.empty() && new_id[new_id.size() - 1] == '.') new_id.erase(new_id.end() - 1);

    // 5단계 처리
    if (new_id.empty()) new_id = "a";

    // 6~7단계 처리
    if (new_id.size() >= 16) {
        new_id = new_id.substr(0, 15);
        if (new_id[new_id.size() - 1] == '.') new_id.erase(new_id.end() - 1);
    }
    else if (new_id.size() <= 2) {
        while (new_id.size() != 3) {
            new_id += new_id[new_id.size() - 1];
        }
    }

    ans = new_id;
    return ans;
}
