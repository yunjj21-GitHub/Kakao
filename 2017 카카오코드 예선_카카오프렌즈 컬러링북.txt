#include <vector>

using namespace std;

vector<vector<int>> painting;
bool chk[100][100];
int M;
int N;
int checking_area;

// 상하좌우 움직임 정의
int dy[4] = { 1, -1, 0, 0 };
int dx[4] = { 0, 0, 1, -1 };

void DFS(int y, int x, int colorNum) {
    if (chk[y][x]) return;

    checking_area++;

    chk[y][x] = true;
    for (int i = 0; i < 4; i++) {
        int nxtY = y + dy[i];
        int nxtX = x + dx[i];

        if (0 > nxtY || nxtY >= M || 0 > nxtX || nxtX >= N) continue;
        if (painting[nxtY][nxtX] != colorNum) continue;

        DFS(nxtY, nxtX, colorNum);
    }
}

// 전역 변수를 정의할 경우 함수 내에 초기화 코드를 꼭 작성해주세요.
vector<int> solution(int m, int n, vector<vector<int>> picture) {
    int number_of_area = 0;
    int max_size_of_one_area = 0;

    painting = picture;
    M = m;
    N = n;

    for (int y = 0; y < m; y++) {
        for (int x = 0; x < n; x++) {
            chk[y][x] = false;
        }
    }

    for (int y = 0; y < m; y++) {
        for (int x = 0; x < n; x++) {
            if (painting[y][x] == 0) continue; // 색칠하지 않는 영역이라면

            if (chk[y][x]) continue; // 이미 방문한 영역이라면

            number_of_area++;
            checking_area = 0;
            DFS(y, x, painting[y][x]);

            if (max_size_of_one_area < checking_area) { // 가장 큰 영역 갱신
                max_size_of_one_area = checking_area;
            }
        }
    }

    // 답안 갱신 
    vector<int> answer(2);
    answer[0] = number_of_area;
    answer[1] = max_size_of_one_area;
    return answer;
}
